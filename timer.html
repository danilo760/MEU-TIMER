Vou criar um código HTML, CSS e JavaScript para um Timer Configurável com as funcionalidades que você solicitou:

<!DOCTYPE html>
<html lang="pt-br">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Timer Configurável</title>
    <style>
        body {
            font-family: Arial, sans-serif;
            display: flex;
            justify-content: center;
            align-items: center;
            height: 100vh;
            background-color: #f5f5f5;
            margin: 0;
        }
        
        .timer-container {
            background-color: white;
            padding: 30px;
            border-radius: 10px;
            box-shadow: 0 4px 8px rgba(0, 0, 0, 0.1);
            text-align: center;
            width: 300px;
        }
        
        h1 {
            margin-top: 0;
            color: #333;
        }
        
        .config-section {
            margin-bottom: 20px;
        }
        
        label {
            display: block;
            margin-bottom: 5px;
            font-weight: bold;
        }
        
        input[type="number"] {
            width: 60px;
            padding: 8px;
            border: 1px solid #ddd;
            border-radius: 4px;
            text-align: center;
        }
        
        .time-display {
            font-size: 48px;
            margin: 20px 0;
            font-family: monospace;
        }
        
        .buttons {
            display: flex;
            justify-content: center;
            gap: 10px;
        }
        
        button {
            padding: 10px 20px;
            border: none;
            border-radius: 4px;
            cursor: pointer;
            font-weight: bold;
            transition: background-color 0.3s;
        }
        
        #startBtn {
            background-color: #4CAF50;
            color: white;
        }
        
        #stopBtn {
            background-color: #f44336;
            color: white;
        }
        
        #resetBtn {
            background-color: #2196F3;
            color: white;
        }
        
        button:hover {
            opacity: 0.9;
        }
        
        button:disabled {
            background-color: #cccccc;
            cursor: not-allowed;
        }
    </style>
</head>
<body>
    <div class="timer-container">
        <h1>Timer Configurável</h1>
        
        <div class="config-section">
            <label for="initialTime">Tempo Inicial (segundos):</label>
            <input type="number" id="initialTime" min="0" value="0">
        </div>
        
        <div class="config-section">
            <label for="addTime">Segundos para Adicionar:</label>
            <input type="number" id="addTime" min="0" value="0">
        </div>
        
        <div class="time-display" id="display">00:00</div>
        
        <div class="buttons">
            <button id="startBtn">Iniciar</button>
            <button id="stopBtn" disabled>Parar</button>
            <button id="resetBtn">Resetar</button>
        </div>
    </div>

    <script>
        document.addEventListener('DOMContentLoaded', function() {
            const initialTimeInput = document.getElementById('initialTime');
            const addTimeInput = document.getElementById('addTime');
            const display = document.getElementById('display');
            const startBtn = document.getElementById('startBtn');
            const stopBtn = document.getElementById('stopBtn');
            const resetBtn = document.getElementById('resetBtn');
            
            let timer;
            let totalSeconds = 0;
            let isRunning = false;
            
            // Formata o tempo para exibição (MM:SS)
            function formatTime(seconds) {
                const mins = Math.floor(seconds / 60);
                const secs = seconds % 60;
                return `${mins.toString().padStart(2, '0')}:${secs.toString().padStart(2, '0')}`;
            }
            
            // Atualiza o display
            function updateDisplay() {
                display.textContent = formatTime(totalSeconds);
            }
            
            // Inicia o timer
            function startTimer() {
                if (isRunning) return;
                
                isRunning = true;
                startBtn.disabled = true;
                stopBtn.disabled = false;
                
                timer = setInterval(function() {
                    if (totalSeconds <= 0) {
                        clearInterval(timer);
                        isRunning = false;
                        startBtn.disabled = false;
                        stopBtn.disabled = true;
                        return;
                    }
                    
                    totalSeconds--;
                    updateDisplay();
                }, 1000);
            }
            
            // Para o timer
            function stopTimer() {
                clearInterval(timer);
                isRunning = false;
                startBtn.disabled = false;
                stopBtn.disabled = true;
            }
            
            // Reseta o timer
            function resetTimer() {
                stopTimer();
                totalSeconds = parseInt(initialTimeInput.value) || 0;
                updateDisplay();
            }
            
            // Adiciona tempo
            function addTime() {
                const secondsToAdd = parseInt(addTimeInput.value) || 0;
                totalSeconds += secondsToAdd;
                updateDisplay();
            }
            
            // Event listeners
            startBtn.addEventListener('click', startTimer);
            stopBtn.addEventListener('click', stopTimer);
            resetBtn.addEventListener('click', resetTimer);
            
            initialTimeInput.addEventListener('change', function() {
                if (!isRunning) {
                    resetTimer();
                }
            });
            
            addTimeInput.addEventListener('change', function() {
                if (parseInt(addTimeInput.value) > 0) {
                    addTime();
                    addTimeInput.value = '0';
                }
            });
            
            // Inicializa
            resetTimer();
        });
    </script>
</body>
</html>

Funcionalidades do Timer:

  1. Configuração Inicial:

    • Você pode definir o tempo inicial em segundos
    • Pode adicionar mais segundos ao tempo atual

  2. Controles:

    • Iniciar: Começa a contagem regressiva
    • Parar: Pausa o timer
    • Resetar: Volta ao tempo inicial configurado

  3. Display:

    • Mostra o tempo no formato MM:SS
    • Atualiza em tempo real durante a contagem

  4. Funcionamento:

    • Quando você altera o tempo inicial e clica em “Resetar”, o timer é atualizado
    • Ao adicionar segundos, o valor é somado ao tempo atual e o campo é resetado para 0

Você pode copiar este código para um arquivo HTML e abri-lo no navegador para usar o timer.
